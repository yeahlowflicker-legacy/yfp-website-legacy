Introduction

This is a Python script that generates desktop entries for applications in Linux systems.

To run the program, make sure to have Python 3 installed. Open terminal in the script directory and type the following command:

python3 desktop-entry-generator.py

Note #0: The program is tested with Python 3.8.

Note #1: The program only generates entries with application type. Other entry types are not supported.

Note #2: Generated files are placed at home/.local/share/applications .

from os.path import exists
from pathlib import Path


def _name():
    name = input("Enter application name: ")
    if len(name) > 0:
        return name
    else:
        print("ERROR: Name cannot be empty.")
        _name()


def _version():
    return input("Enter application version (Skip by entering nothing): ")


def _exec():
    exec = input("Enter executable path (full path): ")
    if exists(exec):
        return exec
    else:
        print("ERROR: Path is invalid.")
        return _exec()


def _icon():
    return input("Enter icon path (full path) (Skip by entering nothing): ")


#   Main Program
if __name__ == "__main__":

    print("Yeahlowflicker Production")
    print("PRJ007 | Linux Desktop Entry Generator")
    print("Press any key to start...")
    input()

    name = _name()
    version = _version()
    exec_path = _exec()
    icon = _icon()

    print("Saving...")

    path = str(Path.home()) + "/.local/share/applications/{}.desktop".format(name.replace(" ", "-"))

    #   Save file
    with open(path, 'a') as file:
        file.write("[Desktop Entry]\n")
        file.write("Type=Application\n")
        file.write("Name={}\n".format(name))
        file.write("Version={}\n".format(version))
        file.write("Exec={}\n".format(exec_path))
        file.write("Icon={}\n".format(icon))

    print("File saved successfully at {}.".format(path))
    print("Press any key to continue...")
